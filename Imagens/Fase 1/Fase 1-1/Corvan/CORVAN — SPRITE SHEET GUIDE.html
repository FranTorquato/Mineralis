🎮 AnimatedSprite2D — Configuração GDScript

# corvan.gd
extends CharacterBody2D

@onready var anim = $AnimatedSprite2D

func _process(delta):
    if velocity.length() > 0:
        anim.play("walk")
    elif is_digging:
        anim.play("dig")
    elif breath_bar < 30:
        anim.play("breathe")  # soroche
    else:
        if has_pickaxe:
            anim.play("idle_tool")
        else:
            anim.play("idle")

func _on_collect_artifact():
    anim.play("victory")
    await anim.animation_finished
    anim.play("idle_tool")
